fetch_latest_salinity <- function(station_info) {
  param <- "sea_water_practical_salinity"
  url <- sprintf(
    "https://erddap.secoora.org/erddap/tabledap/gov-nps-ever-%s.csv?time%%2Clatitude%%2Clongitude%%2C%s&time%%3E=%s&time%%3C=%s",
    station_info$name,
    param,
    start_time,
    end_time
  )

  resp <- request(url) |>
    req_perform()

  df <- read_csv(resp_body_string(resp), show_col_types = FALSE)
  names(df) <- c("time", "latitude", "longitude", "salinity")
  df <- df[-1, ] |>
    mutate(
      time = as.POSIXct(time, format = "%Y-%m-%dT%H:%M:%SZ", tz = "UTC"),
      latitude = as.numeric(latitude),
      longitude = as.numeric(longitude),
      salinity = as.numeric(salinity)
    ) |>
    filter(!is.na(salinity))

  if (nrow(df) == 0) {
    return(tibble(
      station_id = station_info$name,
      station = station_info$long_name,
      location = sprintf(
        "%s (%s)",
        station_info$long_name,
        sub("f1$", "", station_info$name)
      ),
      time = as.POSIXct(NA, origin = "1970-01-01", tz = "UTC"),
      latitude = NA_real_,
      longitude = NA_real_,
      salinity = NA_real_
    ))
  }

  latest <- df[which.max(df$time), ]
  tibble(
    station_id = station_info$name,
    station = station_info$long_name,
    location = sprintf(
      "%s (%s)",
      station_info$long_name,
      sub("f1$", "", station_info$name)
    ),
    time = latest$time,
    latitude = latest$latitude,
    longitude = latest$longitude,
    salinity = latest$salinity
  )
}

salinity_data <- bind_rows(lapply(ndbc_stations, fetch_latest_salinity)) |>

report_time <- format(Sys.time(), "%Y-%m-%d %H:%M %Z")
n_low <- sum(salinity_data$salinity < SALINITY_LOW_THRESHOLD, na.rm = TRUE)
n_high <- sum(salinity_data$salinity > SALINITY_HIGH_THRESHOLD, na.rm = TRUE)
n_normal <- sum(

    salinity_data$salinity >= SALINITY_LOW_THRESHOLD &
    salinity_data$salinity <= SALINITY_HIGH_THRESHOLD
)
n_missing <- sum(is.na(salinity_data$salinity))

Data are pulled from SECOORA ERDDAP using the same 25-hour window as the salinity map. Each gauge shows the latest non-missing reading for an Everglades National Park buoy.

Thresholds

Gauges use practical salinity units (PSU) on a 10–50 scale suited to Florida Bay and the Keys:

Code 
htmltools::tagList(salinity_gauge_legend())
Low: < 25 PSU
Normal: 25–35 PSU
High: > 35 PSU
  • Low (`< 25 PSU): fresher conditions, stronger freshwater or runoff influence
  • Normal (25–35 PSU): typical estuarine range for the region
  • High (`> 35 PSU): more marine or hypersaline conditions
